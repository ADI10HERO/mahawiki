Complete code--

sum = int(input("Enter sum of values to find pair: "))
x = [1,2,5,5,8,0,3,4,10,6]
ans = []
for i in range(0,len(x)-1):
  for j in range(i+1,len(x)):
    if x[i] + x[j] == sum:
      ans.append((x[i],x[j]))
print(ans)
